public Rule datatypeFrame() {
		return sequence(
				"Datatype:",
				Datatype(),
				zeroOrMore(sequence("Annotations:", annotatedList(annotation()))),
				optional(sequence("EquivalentTo:", annotations(), dataRange())),
				zeroOrMore(sequence("Annotations:", annotatedList(annotation()))));
}


public Rule objectPropertyFrame() {
		return sequence(
		        "ObjectProperty:", iriRef(), 
		        zeroOrMore(firstOf(sequence("Annotations:", annotatedList(annotation())),
				sequence("Domain:", annotatedList(description())), sequence(
						"Range:", annotatedList(description())), sequence(
						"Characteristics:",
						annotatedList(objectPropertyCharacteristic())),
				sequence("SubPropertyOf:",
						annotatedList(objectPropertyExpression())), sequence(
						"EquivalentTo:",
						annotatedList(objectPropertyExpression())), sequence(
						"DisjointWith:",
						annotatedList(objectPropertyExpression())),
				sequence("InverseOf:",
						annotatedList(objectPropertyExpression())), sequence(
						"SubPropertyChain:", annotations(), oneOrMore(sequence(
								objectPropertyExpression(), 'o',
								objectPropertyExpression()))))));
}
